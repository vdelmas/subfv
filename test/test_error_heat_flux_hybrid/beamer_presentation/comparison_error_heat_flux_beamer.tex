\begin{frame}{Compared schemes~\cite{Delmas2025, Delmas2026}}
  All schemes: subface-based, cell-centered FV approximate Riemann solvers.
  Same positivity-preserving wave speeds; differ in contact wave / tangential velocity treatment.
  \bigskip
  \begin{description}
    \item[\texttt{two\_wave}] (HLL-like) — single star state; over-diffuses shear layers
      $\Rightarrow$ over-estimates wall heat flux
    \item[\texttt{three\_wave}] (HLLC-like) — exact contact resolution; \alert{carbuncle} on hybrid meshes
    \item[\texttt{modified\_three\_wave}] (HLLCM-like) — stable contact, averaged tangential velocity;
      residual diffusion on BL cells prevents convergence of $C_h$
    \item[\texttt{multi\_point\_iso}] (Isotropic multi-point) — node-conservative, isotropic weights;
      \textbf{stable + accurate} on anisotropic hybrid meshes
  \end{description}
\end{frame}

\begin{frame}{Summary}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Key findings}
      \begin{itemize}
        \item \texttt{two\_wave}: always converges but large $C_h$ error
        \item \texttt{three\_wave}: carbuncle instabilities for fine BL
        \item \texttt{modified\_three\_wave}: stable but $C_h$ does not converge with BL refinement
        \item \texttt{multi\_point\_iso}: \textbf{stable and accurate} for all parameter combinations
      \end{itemize}
    \end{column}
